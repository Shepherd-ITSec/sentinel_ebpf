# Train LODA on the train split and save a checkpoint.
uv run python -m scripts.train_detector_checkpoint \
  "/home/felix/sentinel_ebpf/test_data/lidds/cve-2012-2122.train.jsonl" \
  --algorithm loda \
  --score-dump "/home/felix/sentinel_ebpf/test_data/lidds/loda_train_scores.jsonl" \
  --out "/home/felix/sentinel_ebpf/test_data/lidds/loda_training.pkl"

uv run python -m scripts.score_from_checkpoint \
  "/home/felix/sentinel_ebpf/test_data/lidds/cve-2012-2122.test.jsonl" \
  --algorithm loda \
  --checkpoint "/home/felix/sentinel_ebpf/test_data/lidds/loda_training.pkl" \
  --out "/home/felix/sentinel_ebpf/test_data/lidds/loda_test_scores_13r.jsonl" \
  --max-recordings 13
