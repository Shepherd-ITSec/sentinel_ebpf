# Retest updated MemStream and LODA-EMA on the first 500k events of events_17_03_26.jsonl.
# Run with:
#   ./scripts/run_commands_overnight.sh -f scripts/commands_memstream_loda_17_500k.txt memstream_loda_17_500k.log

mkdir -p test_data/view_retest_17_500k

# MemStream in the modes we care about (freq1d_z_surprisal excluded; doesn't work well).

# LODA-EMA with its current algorithm-selected feature view.
uv run python scripts/model_diagnostic.py --algorithm loda_ema --limit 500000 --out-dir test_data/view_retest_17_500k/loda_ema events_17_03_26.jsonl
